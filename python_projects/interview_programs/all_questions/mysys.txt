# find items b which are not in a:

b = ['r', 's', 'p', 't', 'u', 'v', 'w', 'x', 'y', 'z']

result= [item for item in b if item not in a.values()]
print("".join(result))

# Class to find area of triangle
class area:
    def __init__(self, b,h):
        self.b= b
        self.h= h

    def find_area(self):
        area= (0.5*self.b*self.h)
        print(area)


obj= area(2, 2)
