- name: Play1
  hosts: host_name
  vars:
    pkg:
      - httpd
      - nginx
    age: 20
  tasks:
    - name: Task1 to install packages
      apt:
        name: "{{ item }}"
        state: latest or present
      with_items: "{{ pkg }}"
      when: age > 20

    - name: Task2 to enable the service
      service:
        name: "{{ item }}"
        state: started
      with_items: {{ pkg }}
      when: age>20
